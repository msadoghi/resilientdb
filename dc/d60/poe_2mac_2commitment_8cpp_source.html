103int Commitment::Process(int type, std::unique_ptr<POERequest> request) {

125void Commitment::StartNewView(std::unique_ptr<Request> user_request) {
126 auto seq = message_manager_->AssignNextSeq();
127 if (!seq.ok()) {
128 global_stats_->SeqFail();
129 SendFailRepsonse(*user_request);
130 return;
131 }
132
133 std::unique_ptr<POERequest> poe_request =
134 TransactionCollector::ConvertToCustomRequest<POERequest>(*user_request);
135
136 poe_request->set_sender_id(id_);
137 poe_request->set_seq(*seq);
138 poe_request->set_current_view(message_manager_->GetCurrentView());
139 poe_request->set_type(POERequest::TYPE_SUPPORT);
140
141 client_call_(POERequest::TYPE_SUPPORT, *poe_request, -1);
142}

144int Commitment::ProcessSupport(std::unique_ptr<POERequest> request) {
145 global_stats_->IncPropose();
146 uint64_t curent_view = message_manager_->GetCurrentView();
147
148 if (PrimaryId(curent_view) != request->sender_id()) {
149 LOG(ERROR) << "not send from primary:";
150 return -2;
151 }
152
153 CollectorResultCode ret = message_manager_->AddConsensusMsg(
154 TransactionCollector::ConvertToRequest(*request));
155 if (ret == CollectorResultCode::STATE_CHANGED) {
156 POERequest poe_request(*request);
157 poe_request.set_sender_id(id_);
158 poe_request.set_type(POERequest::TYPE_CERTIFY);
159 poe_request.clear_data();
160 client_call_(POERequest::TYPE_CERTIFY, poe_request, -1);
161 }
162 return ret == CollectorResultCode::INVALID ? -2 : 0;
163}

165int Commitment::ProcessCeritify(std::unique_ptr<POERequest> request) {
166 global_stats_->IncCommit();
167 CollectorResultCode ret = message_manager_->AddConsensusMsg(
168 TransactionCollector::ConvertToRequest(*request));
169 return ret == CollectorResultCode::INVALID ? -2 : 0;

172// =========== private threads ===========================
173// If the transaction is executed, send back to the proxy.
174int Commitment::PostProcessExecutedMsg() {
175 while (!stop_) {
176 auto batch_resp = message_manager_->GetResponseMsg();
177 if (batch_resp == nullptr) {
178 continue;
179 }
180 Request request;
181 request.set_seq(batch_resp->seq());
182 request.set_type(Request::TYPE_RESPONSE);
183 request.set_sender_id(config_.GetSelfInfo().id());
184 request.set_current_view(batch_resp->current_view());
185 request.set_proxy_id(batch_resp->proxy_id());
186 batch_resp->SerializeToString(request.mutable_data());
187 client_call_(0, request, request.proxy_id());
188 }
189 return 0;
190}
