90int ConsensusManagerPOE::ConsensusCommit(std::unique_ptr<Context> context,
91 std::unique_ptr<Request> request) {
92 // LOG(ERROR) << "get request:" << Request::Type_Name(request->type())
93 // << " from:" << request->sender_id();
94 switch (request->type()) {
95 case Request::TYPE_CLIENT_REQUEST:
96 if (config_.IsPerformanceRunning()) {
97 return performance_manager_->StartEval();
98 }
99 return response_manager_->NewUserRequest(std::move(context),
100 std::move(request));
101 case Request::TYPE_RESPONSE:
102 if (config_.IsPerformanceRunning()) {
103 return performance_manager_->ProcessResponseMsg(std::move(context),
104 std::move(request));
105 }
106 return response_manager_->ProcessResponseMsg(std::move(context),
107 std::move(request));
108 case Request::TYPE_VIEWCHANGE:
109 return viewchange_manager_->ProcessViewChange(
110 DecodeMsg<ViewChangeMsg>(*request));
111 case Request::TYPE_NEWVIEW:
112 return viewchange_manager_->ProcessNewView(
113 DecodeMsg<NewViewChangeMsg>(*request));
114 default:
115 return commitment_->Process(Decode(*request));
116 }
117 return 0;
118}
