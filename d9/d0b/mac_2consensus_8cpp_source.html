83int Consensus::ConsensusCommit(std::unique_ptr<Context> context,
84 std::unique_ptr<Request> request) {
85 // LOG(ERROR) << "get request:" << Request::Type_Name(request->type())
86 // << " from:" << request->sender_id()<<" seq:"<<request->seq();
87 switch (request->type()) {
88 case Request::TYPE_CLIENT_REQUEST:
89 if (config_.IsPerformanceRunning()) {
90 return performance_manager_->StartEval();
91 }
92 return response_manager_->NewUserRequest(std::move(context),
93 std::move(request));
94 case Request::TYPE_RESPONSE:
95 if (config_.IsPerformanceRunning()) {
96 return performance_manager_->ProcessResponseMsg(std::move(context),
97 std::move(request));
98 }
99 return response_manager_->ProcessResponseMsg(std::move(context),
100 std::move(request));
101 case Request::TYPE_CUSTOM_TYPE:
102 return commitment_->Process(request->user_type(), Decode(*request));
103
104 case Request::TYPE_NEW_TXNS:
105 return commitment_->ProcessNewTransaction(std::move(request));
106 default:
107 return commitment_->Process(request->user_type(), Decode(*request));
108 }
109 return 0;
110}

116int Consensus::SendMessage(int user_type,
117 const ::google::protobuf::Message& msg,
118 int proxy_id) {
119 if (user_type == 0) {
120 communicator_->SendMessage(msg, proxy_id);
121 return 0;
122 } else {
123 Request request;
124 request.set_user_type(user_type);
125 request.set_type(Request::TYPE_CUSTOM_TYPE);
126 request.set_sender_id(config_.GetSelfInfo().id());
127 msg.SerializeToString(request.mutable_data());
128 if (proxy_id == -1) {
129 communicator_->BroadCast(request);
130 } else {
131 communicator_->SendMessage(request, proxy_id);
132 }
133 }
134 return 0;
135}
