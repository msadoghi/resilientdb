66 template <typename T>
67 class AtomicUniquePtr {
68 public:
69 AtomicUniquePtr() : v_(0) {}
70 bool Set(std::unique_ptr<T>& new_ptr) {
71 int old_v = 0;
72 bool res = v_.compare_exchange_strong(old_v, 1, std::memory_order_acq_rel,
73 std::memory_order_acq_rel);
74 if (!res) {
75 return false;
76 }
77 ptr_ = std::move(new_ptr);
78 v_ = 2;
79 return true;
80 }
81
82 T* Reference() {
83 int v = v_.load(std::memory_order_acq_rel);
84 if (v <= 1) {
85 return nullptr;
86 }
87 return ptr_.get();
88 }
89
90 private:
91 std::unique_ptr<T> ptr_;
92 std::atomic<int> v_;
93 };
