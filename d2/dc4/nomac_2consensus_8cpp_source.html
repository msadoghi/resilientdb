82int Consensus::ConsensusCommit(std::unique_ptr<Context> context,
83 std::unique_ptr<Request> request) {
84 // LOG(ERROR) << "get request:" << Request::Type_Name(request->type())
85 // << " from:" << request->sender_id();
86 switch (request->type()) {
87 case Request::TYPE_CLIENT_REQUEST:
88 if (config_.IsPerformanceRunning()) {
89 return performance_manager_->StartEval();
90 }
91 return response_manager_->NewUserRequest(std::move(context),
92 std::move(request));
93 case Request::TYPE_RESPONSE:
94 if (config_.IsPerformanceRunning()) {
95 return performance_manager_->ProcessResponseMsg(std::move(context),
96 std::move(request));
97 }
98 return response_manager_->ProcessResponseMsg(std::move(context),
99 std::move(request));
100 case Request::TYPE_NEW_TXNS:
101 return commitment_->ProcessNewTransaction(std::move(request));
102 default:
103 return commitment_->Process(request->user_type(), Decode(*request));
104 }
105 return 0;
106}

112int Consensus::SendMessage(int user_type,
113 const ::google::protobuf::Message& msg,
114 int proxy_id) {
115 if (user_type == 0) {
116 communicator_->SendMessage(msg, proxy_id);
117 return 0;
118 } else {
119 Request request;
120 request.set_user_type(user_type);
121 request.set_type(Request::TYPE_CUSTOM_TYPE);
122 request.set_sender_id(config_.GetSelfInfo().id());
123 msg.SerializeToString(request.mutable_data());
124 if (proxy_id == -1) {
125 communicator_->BroadCast(request);
126 } else {
127 communicator_->SendMessage(request, proxy_id);
128 }
129 }
130 return 0;
131}
