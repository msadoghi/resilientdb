130void Commitment::StartNewView(std::unique_ptr<Request> user_request) {
131 // LOG(ERROR) << "start primary id:" << PrimaryId(current_view_) << " "
132 // << current_view_;
133 if (PrimaryId(message_manager_->GetCurrentView()) != id_) {
134 LOG(ERROR) << "not a primary, redirect to:"
135 << message_manager_->GetCurrentView();
136 RedirectUserRequest(*user_request);
137 return;
138 }
139
140 if (message_manager_->IsLockCurrentView()) {
141 LOG(ERROR) << " view is lock:" << message_manager_->GetCurrentView();
142 SendFailRepsonse(*user_request);
143 return;
144 }
145
146 auto seq = message_manager_->AssignNextSeq();
147 if (!seq.ok()) {
148 global_stats_->SeqFail();
149 SendFailRepsonse(*user_request);
150 return;
151 }
152
153 std::unique_ptr<POERequest> poe_request =
154 ConvertToCustomRequest(*user_request);
155
156 poe_request->set_type(POERequest::TYPE_PROPOSE);
157 poe_request->set_seq(*seq);
158 poe_request->set_current_view(message_manager_->GetCurrentView());
159 poe_request->set_sender_id(id_);
160 *poe_request->mutable_data_signature() = user_request->data_signature();
161 client_call_(POERequest::TYPE_PROPOSE, *poe_request, -1);
162}

164bool Commitment::VerifyNodeSignagure(const POERequest& request) {
165 if (verifier_ == nullptr) {
166 return true;
167 }
168
169 if (!request.has_data_signature()) {
170 LOG(ERROR) << "node signature is empty";
171 return false;
172 }
173 // check signatures
174 bool valid =
175 verifier_->VerifyMessage(request.hash(), request.data_signature());
176 if (!valid) {
177 LOG(ERROR) << "signature is not valid:"
178 << request.data_signature().DebugString();
179 return false;
180 }
181 return true;
182}

184bool Commitment::VerifyTS(const POERequest& request) {
185 if (verifier_ == nullptr) {
186 return true;
187 }
188 if (request.qc().signatures_size() < config_.GetMinDataReceiveNum()) {
189 LOG(ERROR) << "signature is not enough:" << request.qc().signatures_size()
190 << " seq:" << request.seq();
191 return false;
192 }
193 for (const auto& signature : request.qc().signatures()) {
194 bool valid = verifier_->VerifyMessage(request.hash(), signature);
195 if (!valid) {
196 LOG(ERROR) << "TS is not valid:" << request.seq();
197 return false;
198 }
199 }
200 return true;
201}

203bool Commitment::VerifyTxn(const POERequest& request) {
204 // Verify the client signature
205 if (request.sender_id() != config_.GetSelfInfo().id()) {
206 // check signatures
207 bool valid =
208 verifier_->VerifyMessage(request.data(), request.data_signature());
209 if (!valid) {
210 LOG(ERROR) << "request is not valid:"
211 << request.data_signature().DebugString();
212 LOG(ERROR) << " msg:" << request.data().size();
213 return false;
214 }
215 }
216 return true;
217}

219int Commitment::ProcessMessageOnPrimary(std::unique_ptr<POERequest> request) {
220 // POERequest::Type type = (POERequest::Type)request->type();
221 // LOG(INFO) << "primary get type:" << POERequest_Type_Name(type)
222 // << " view:" << request->current_view() << " seq:" <<
223 // request->seq()
224 // << " from:" << request->sender_id()
225 // << " request:" << request->DebugString();
226 if (!VerifyNodeSignagure(*request)) {
227 LOG(ERROR) << "signature invalid";
228 return -2;
229 }
230
231 std::unique_ptr<Request> req = ConvertToRequest(*request);
232
233 if (request->has_data_signature()) {
234 *req->mutable_data_signature() = request->data_signature();
235 }
236
237 CollectorResultCode ret = message_manager_->AddConsensusMsg(std::move(req));
238 if (ret == CollectorResultCode::STATE_CHANGED) {
239 POERequest poe_request(*request);
240 *poe_request.mutable_qc() = message_manager_->GetQC(request->seq());
241 assert(poe_request.qc().signatures_size() > 0);
242
243 poe_request.set_sender_id(id_);
244 poe_request.set_type(POERequest::TYPE_CERTIFY);
245 client_call_(POERequest::TYPE_CERTIFY, poe_request, -1);
246 message_manager_->Clear(poe_request.seq());
247 }
248 return 0;
249}

251int Commitment::ProcessMessageOnReplica(std::unique_ptr<POERequest> request) {
252 // LOG(ERROR) << "Replica receive type:" <<
253 // POERequest_Type_Name(request->type())
254 // << " from:" << request->sender_id() << " seq:" << request->seq();
255 if (request->type() == POERequest::TYPE_PROPOSE) {
256 global_stats_->IncPropose();
257 }
258 if (request->type() == POERequest::TYPE_CERTIFY) {
259 global_stats_->IncCommit();
260 }
261 if (request->type() == POERequest::TYPE_PROPOSE) {
262 uint64_t curent_view = message_manager_->GetCurrentView();
263 if (request->current_view() != curent_view) {
264 LOG(ERROR) << "current view is not the same:" << request->current_view()
265 << " current view:" << curent_view;
266 return -2;
267 }
268 if (PrimaryId(curent_view) != request->sender_id()) {
269 LOG(ERROR) << "not send from primary:";
270 return -2;
271 }
272 // check if has been received.
273 if (message_manager_->IsCommitted(request->seq())) {
274 LOG(ERROR) << "current seq is larger:" << request->seq();
275 return -2;
276 }
277 // Verify the client signature
278 if (!VerifyTxn(*request)) {
279 return -2;
280 }
281 } else {
282 if (!VerifyTS(*request)) {
283 LOG(ERROR) << " ts not invlid"
284 << " request type:" << POERequest_Type_Name(request->type())
285 << " seq:" << request->seq();
286 return -2;
287 }
288 }
289
290 // LOG(INFO) << "Replica receive type:" <<
291 // POERequest_Type_Name(request->type())
292 // << " view:" << request->view() << " seq:" << request->seq()
293 // << " from:" << request->sender_id();
294 if (request->type() == POERequest::TYPE_CERTIFY) {
295 // execute and send new view
296 message_manager_->Commit(std::move(request));
297 } else {
298 POERequest poe_request(*request);
299 int sender = poe_request.sender_id();
300 poe_request.set_hash(GetHash(*request));
301 poe_request.clear_data();
302 if (verifier_) {
303 auto signature_or = verifier_->SignMessage(poe_request.hash());
304 if (!signature_or.ok()) {
305 LOG(ERROR) << "Sign message fail";
306 return -2;
307 }
308 *poe_request.mutable_data_signature() = *signature_or;
309 }
310 request->set_hash(poe_request.hash());
311 message_manager_->SavePreparedData(std::move(request));
312
313 // LOG(INFO) << "send type:"
314 // << POERequest_Type_Name(VoteType(poe_request.type()))
315 // << " to:" << poe_request.sender_id() << " view"
316 // << poe_request.current_view() << " seq:" << poe_request.seq();
317 poe_request.set_type(VoteType(poe_request.type()));
318 poe_request.set_sender_id(id_);
319 client_call_(poe_request.type(), poe_request, sender);
320 }
321 return 0;
322}

324// =========== private threads ===========================
325// If the transaction is executed, send back to the proxy.
326int Commitment::PostProcessExecutedMsg() {
327 while (!stop_) {
328 auto batch_resp = message_manager_->GetResponseMsg();
329 if (batch_resp == nullptr) {
330 continue;
331 }
332 Request request;
333 request.set_seq(batch_resp->seq());
334 request.set_type(Request::TYPE_RESPONSE);
335 request.set_sender_id(config_.GetSelfInfo().id());
336 request.set_current_view(batch_resp->current_view());
337 request.set_proxy_id(batch_resp->proxy_id());
338 batch_resp->SerializeToString(request.mutable_data());
339 client_call_(0, request, request.proxy_id());
340 }
341 return 0;
342}
