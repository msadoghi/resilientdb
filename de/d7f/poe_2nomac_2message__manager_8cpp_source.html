96void MessageManager::CommitInternal(const POERequest& poe_request,
97 const std::string& data) {
98 if (system_info_->GetCurrentView() != poe_request.current_view()) {
99 LOG(ERROR) << " VIEW NOT MATCH, SKIP COMMIT";
100 return;
101 }
102 if (poe_request.current_view() <= view_locked_) {
103 LOG(ERROR) << " view:" << poe_request.current_view()
104 << " is locked, rejected:" << view_locked_;
105 return;
106 }
107 std::unique_ptr<Request> execute_request = std::make_unique<Request>();
108 execute_request->set_data(data);
109 execute_request->set_seq(poe_request.seq());
110 execute_request->set_proxy_id(poe_request.proxy_id());
111 // LOG(ERROR) << "commit seq:" << poe_request.seq()
112 // << " data size:" << execute_request->data().size();
113 committed_seq_.insert(poe_request.seq());
114 transaction_executor_->Commit(std::move(execute_request));
115}

121void MessageManager::SetRequest(std::unique_ptr<POERequest> poe_request) {
122 std::string hash = poe_request->hash();
123 int64_t seq = poe_request->seq();
124
125 std::lock_guard<std::mutex> lk(mutex_);
126 if (poe_request->type() == POERequest::TYPE_CERTIFY) {
127 commit_req_[seq] = std::move(poe_request);
128 } else if (poe_request->type() == POERequest::TYPE_PROPOSE) {
129 data_[hash] = poe_request->data();
130 }
131 if (commit_req_.find(seq) != commit_req_.end() &&
132 data_.find(hash) != data_.end()) {
133 CommitInternal(*commit_req_[seq], data_[hash]);
134 }
135}

146CertifyRequests MessageManager::GetCertifyRequests() {
147 std::lock_guard<std::mutex> lk(mutex_);
148 CertifyRequests msg;
149 for (uint64_t seq = 1;; seq++) {
150 if (commit_req_.find(seq) == commit_req_.end()) {
151 break;
152 }
153 POERequest request = *commit_req_[seq];
154 if (data_.find(request.hash()) == data_.end()) {
155 break;
156 }
157 request.set_data(data_[request.hash()]);
158 *msg.add_certify_requests() = request;
159 }
160 LOG(ERROR) << "get data seq:" << msg.certify_requests_size();
161 return msg;
162}

168bool MessageManager::MayConsensusChangeStatus(
169 int type, int received_count, std::atomic<TransactionStatue>* status) {
170 switch (type) {
171 case POERequest::TYPE_SUPPORT:
172 if (*status == TransactionStatue::None &&
173 config_.GetMinDataReceiveNum() <= received_count) {
174 TransactionStatue old_status = TransactionStatue::None;
175 return status->compare_exchange_strong(
176 old_status, TransactionStatue::EXECUTED, std::memory_order_acq_rel,
177 std::memory_order_acq_rel);
178 }
179 break;
180 }
181 return false;
182}

184CollectorResultCode MessageManager::AddConsensusMsg(
185 std::unique_ptr<Request> request) {
186 if (request == nullptr) {
187 return CollectorResultCode::INVALID;
188 }
189 int type = request->type();
190 uint64_t seq = request->seq();
191 int resp_received_count = 0;
192 int ret = collector_pool_->GetCollector(seq)->AddRequest(
193 std::move(request), false,
194 [&](const Request& request, int received_count,
195 std::atomic<TransactionStatue>* status) {
196 if (MayConsensusChangeStatus(type, received_count, status)) {
197 resp_received_count = 1;
198 }
199 });
200 if (ret != 0) {
201 return CollectorResultCode::INVALID;
202 }
203 if (resp_received_count > 0) {
204 return CollectorResultCode::STATE_CHANGED;
205 }
206 return CollectorResultCode::OK;
207}
