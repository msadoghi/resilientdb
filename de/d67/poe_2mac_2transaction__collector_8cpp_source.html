37int TransactionCollector::AddRequest(
38 std::unique_ptr<Request> request, bool is_main_request,
39 std::function<void(const Request&, int received_count,
40 std::atomic<TransactionStatue>* status)>
41 call_back) {
42 if (request == nullptr) {
43 LOG(ERROR) << "request empty";
44 return -2;
45 }
46
47 int32_t sender_id = request->sender_id();
48 std::string hash = request->hash();
49 int type = request->type();
50 uint64_t seq = request->seq();
51 if (is_committed_) {
52 return -2;
53 }
54 if (status_.load() == EXECUTED) {
55 return -2;
56 }
57
58 if (seq_ != static_cast<uint64_t>(request->seq())) {
59 LOG(ERROR) << "data invalid, seq not the same:" << seq
60 << " collect seq:" << seq_;
61 return -2;
62 }
63
64 if (is_main_request) {
65 auto request_info = std::make_unique<RequestInfo>();
66 request_info->request = std::move(request);
67 int ret = atomic_mian_request_.Set(request_info);
68 if (!ret) {
69 LOG(ERROR) << "set main request fail: data existed:" << seq
70 << " ret:" << ret;
71 return -2;
72 }
73 auto main_request = atomic_mian_request_.Reference();
74 if (main_request->request == nullptr) {
75 LOG(ERROR) << "set main request data fail";
76 return -2;
77 }
78 call_back(*main_request->request.get(), 1, &status_);
79 return 0;
80 } else {
81 senders_[type][sender_id] = 1;
82 call_back(*request, senders_[type].count(), &status_);
83 if (status_.load() == TransactionStatue::READY_EXECUTE) {
84 Commit();
85 }
86 }
87 return 0;
88}

90int TransactionCollector::Commit() {
91 TransactionStatue old_status = TransactionStatue::READY_EXECUTE;
92 bool res = status_.compare_exchange_strong(
93 old_status, TransactionStatue::EXECUTED, std::memory_order_acq_rel,
94 std::memory_order_acq_rel);
95 if (!res) {
96 return -2;
97 }
98
99 auto main_request = atomic_mian_request_.Reference();
100 if (main_request == nullptr) {
101 LOG(ERROR) << "no main";
102 return -2;
103 }
104
105 is_committed_ = true;
106 if (executor_ && main_request->request) {
107 executor_->Commit(std::move(main_request->request));
108 }
109 return 0;
110}
