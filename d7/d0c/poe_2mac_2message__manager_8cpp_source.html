80bool MessageManager::MayConsensusChangeStatus(
81 int type, int received_count, std::atomic<TransactionStatue>* status) {
82 switch (type) {
83 case POERequest::TYPE_SUPPORT:
84 if (*status == TransactionStatue::None) {
85 TransactionStatue old_status = TransactionStatue::None;
86 return status->compare_exchange_strong(
87 old_status, TransactionStatue::READY_COMMIT,
88 std::memory_order_acq_rel, std::memory_order_acq_rel);
89 }
90 break;
91 case POERequest::TYPE_CERTIFY:
92 if (*status == TransactionStatue::READY_COMMIT &&
93 config_.GetMinDataReceiveNum() <= received_count) {
94 TransactionStatue old_status = TransactionStatue::READY_COMMIT;
95 return status->compare_exchange_strong(
96 old_status, TransactionStatue::READY_EXECUTE,
97 std::memory_order_acq_rel, std::memory_order_acq_rel);
98 }
99 break;
100 }
101 return false;
102}

104bool MessageManager::IsValidMsg(const Request& request) {
105 // view should be the same as the current one.
106 if (static_cast<uint64_t>(request.current_view()) != GetCurrentView()) {
107 LOG(ERROR) << "message view :[" << request.current_view()
108 << "] is older than the cur view :[" << GetCurrentView() << "]";
109 return false;
110 }
111
112 if (static_cast<uint64_t>(request.seq()) <
113 transaction_executor_->GetMaxPendingExecutedSeq()) {
114 return false;
115 }
116
117 return true;
118}

120CollectorResultCode MessageManager::AddConsensusMsg(
121 std::unique_ptr<Request> request) {
122 if (request == nullptr || !IsValidMsg(*request)) {
123 return CollectorResultCode::INVALID;
124 }
125 int type = request->type();
126 uint64_t seq = request->seq();
127 int resp_received_count = 0;
128 int ret = collector_pool_->GetCollector(seq)->AddRequest(
129 std::move(request), type == POERequest::TYPE_SUPPORT,
130 [&](const Request& request, int received_count,
131 std::atomic<TransactionStatue>* status) {
132 if (MayConsensusChangeStatus(type, received_count, status)) {
133 resp_received_count = 1;
134 }
135 });
136 if (ret != 0) {
137 return CollectorResultCode::INVALID;
138 }
139 if (resp_received_count > 0) {
140 return CollectorResultCode::STATE_CHANGED;
141 }
142 return CollectorResultCode::OK;
143}
