81bool ViewChange::VerifyTS(const NewViewChangeMsg& request) {
82 if (verifier_ == nullptr) {
83 return true;
84 }
85 if (request.qc().signatures_size() < config_.GetMinDataReceiveNum()) {
86 LOG(ERROR) << "signature is not enough:" << request.qc().signatures_size();
87 return false;
88 }
89 for (const auto& signature : request.qc().signatures()) {
90 bool valid =
91 verifier_->VerifyMessage(std::to_string(request.view()), signature);
92 if (!valid) {
93 LOG(ERROR) << "TS is not valid";
94 return false;
95 }
96 }
97 return true;
98}

100bool ViewChange::VerifyNodeSignagure(const POERequest& request) {
101 if (verifier_ == nullptr) {
102 return true;
103 }
104
105 if (!request.has_hash_signature()) {
106 LOG(ERROR) << "node signature is empty";
107 return false;
108 }
109 // check signatures
110 bool valid =
111 verifier_->VerifyMessage(request.hash(), request.hash_signature());
112 if (!valid) {
113 LOG(ERROR) << "signature is not valid:"
114 << request.hash_signature().DebugString();
115 return false;
116 }
117 return true;
118}

137 while (!is_stop_) {

194void ViewChange::SendNewView(uint64_t new_view) {
195 NewViewChangeMsg msg;
196
197 // Obtain the valid certify requets.
198 std::map<int64_t, POERequest> certify_reqs;
199 for (const auto& request : new_view_requests_[new_view]) {
200 for (const auto& vc_request :
201 request->certify_requests().certify_requests()) {
202 if (certify_reqs.find(vc_request.seq()) != certify_reqs.end()) {
203 continue;
204 }
205 if (VerifyNodeSignagure(vc_request)) {
206 certify_reqs[vc_request.seq()] = vc_request;
207 }
208 }
209 }
210 for (uint64_t seq = 1;; seq++) {
211 if (certify_reqs.find(seq) == certify_reqs.end()) {
212 break;
213 }
214 *msg.mutable_certify_requests()->add_certify_requests() = certify_reqs[seq];
215 }
216
217 auto request = std::make_unique<Request>();
218 *msg.mutable_qc() = GetQC(new_view);
219 msg.set_view(new_view);
220 msg.set_sender_id(config_.GetSelfInfo().id());
221 msg.SerializeToString(request->mutable_data());
222
223 request->set_type(Request::TYPE_NEWVIEW);
224 LOG(ERROR) << "Send vc to primary:"
225 << PrimaryId(new_view, config_.GetReplicaInfos().size());
226 replica_communicator_->BroadCast(*request);
227}

229int ViewChange::ProcessViewChange(std::unique_ptr<ViewChangeMsg> vc_msg) {
230 std::unique_lock<std::mutex> lk(mutex_);
231 uint64_t new_view = vc_msg->view();
232 LOG(INFO) << "recv vc request view:" << new_view
233 << " from:" << vc_msg->sender_id()
234 << " total:" << new_view_msgs_[new_view].size();
235 if (new_view_msgs_[new_view].find(vc_msg->sender_id()) !=
236 new_view_msgs_[new_view].end()) {
237 return 0;
238 }
239 if (new_view <= system_info_->GetCurrentView()) {
240 LOG(ERROR) << "new view:" << new_view << " out dated";
241 return -2;
242 }
243 new_view_msgs_[new_view].insert(vc_msg->sender_id());
244 new_view_requests_[new_view].push_back(std::move(vc_msg));
245 // f+1
246 if (new_view_msgs_[new_view].size() ==
247 static_cast<size_t>(config_.GetMaxMaliciousReplicaNum()) + 1) {
248 if (new_view_msgs_[new_view].find(config_.GetSelfInfo().id()) ==
249 new_view_msgs_[new_view].end()) {
250 // received f+1 vc msg.
251 TriggerViewChange(new_view);
252 }
253 }
254 // 2f+1
255 if (new_view_msgs_[new_view].size() ==
256 static_cast<size_t>(config_.GetMinDataReceiveNum())) {
257 if (PrimaryId(new_view, config_.GetReplicaInfos().size()) ==
258 config_.GetSelfInfo().id()) {
259 // SendNewView
260 SendNewView(new_view);
261 }
262 }
263 return 0;
264}
