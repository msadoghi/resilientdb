33int QCCollector::AddRequest(
34 std::unique_ptr<Request> request, bool is_main_request,
35 std::function<void(const Request&, int received_count,
36 std::atomic<TransactionStatue>* status)>
37 call_back) {
38 if (request == nullptr) {
39 LOG(ERROR) << "request empty";
40 return -2;
41 }
42
43 int32_t sender_id = request->sender_id();
44 std::string hash = request->hash();
45 int type = request->type();
46 uint64_t seq = request->seq();
47 if (is_committed_) {
48 return -2;
49 }
50 if (status_.load() == EXECUTED) {
51 return -2;
52 }
53
54 if (seq_ != static_cast<uint64_t>(request->seq())) {
55 // LOG(ERROR) << "data invalid, seq not the same:" << seq
56 // << " collect seq:" << seq_;
57 return -2;
58 }
59
60 if (is_main_request) {
61 auto request_info = std::make_unique<RequestInfo>();
62 request_info->request = std::move(request);
63 int ret = atomic_mian_request_.Set(request_info);
64 if (!ret) {
65 LOG(ERROR) << "set main request fail: data existed:" << seq
66 << " ret:" << ret;
67 return -2;
68 }
69 auto main_request = atomic_mian_request_.Reference();
70 if (main_request->request == nullptr) {
71 LOG(ERROR) << "set main request data fail";
72 return -2;
73 }
74 call_back(*main_request->request.get(), 1, &status_);
75 return 0;
76 } else {
77 if (request->has_data_signature() &&
78 request->data_signature().node_id() > 0) {
79 std::lock_guard<std::mutex> lk(mutex_);
80 *commit_certs_.add_signatures() = request->data_signature();
81 } else {
82 LOG(ERROR) << "no sig ===== " << seq_;
83 }
84 senders_[type][sender_id] = 1;
85 call_back(*request, senders_[type].count(), &status_);
86 }
87 return 0;
88}
